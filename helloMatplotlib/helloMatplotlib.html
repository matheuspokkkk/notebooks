#o primeiro plot tem como os valores de x e y os valores do vetor
#o primeiro é representado por red-- (tracinhos vermelhos)

#o segundo é representado por tracinhos azuis

#posso imprimir diversos plots juntos em um único gráfico

Plotando vários gráficos em uma única imagem (subplots):¶

In [5]:
bigPicture, smallPics = matplotlib.pyplot.subplots(nrows = 2, ncols = 2)
#bigPicture == variável que cuida da "imagem geral", que conterá todos os gráficos.
#smallPicss == variável que armazena a lista de gráficos.
#para mexer na imagem geral, mexo na variável bigPicture, e para mexer em gráficos individuais, na smallPics
#no fim, digo que a imagem terá dois gráficos em cada coluna e também dois gráficos em cada linha

vector = numpy.array([0, 1, 2, 3, 4, 5, 6])
#novamente criando um vetor

smallPics[0][0].hist(vector, vector)
smallPics[0][1].plot(vector, vector)
smallPics[1][0].plot(vector, vector, 'ro')
smallPics[1][1].scatter(vector, vector)
#montando o gráfico da linha 0, coluna 0.. será um histograma
#montando o gráfico da linha 0, coluna 1, será um plot normal
#montando o gráfico da linha 1, coluna 0: plot normal cujos pontos são bolinhas vermelhas
#montando linha 1, coluna 1: gráfico de dispersão

bigPicture.suptitle("Titulo da Imagem Geral")
#atribui um titulo à imagem que contem os gráficos

matplotlib.pyplot.show()
#imprime o gráfico na tela

matplotlib.pyplot.savefig("nomeDoArquivo.png")
#salva a imagem com os plots no diretorio que que o código for executado, com o nome do arquivo estipulado
